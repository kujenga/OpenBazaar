10  """ This module defines the charaterizing constants of the Kademlia network 
11   
12  C{checkRefreshInterval} and C{udpDatagramMaxSize} are implementation-specific 
13  constants, and do not affect general Kademlia operation.  
14  """ 
15   
16  ######### KADEMLIA CONSTANTS ########### 
17   
18  #: Small number Representing the degree of parallelism in network calls 
19  alpha = 3 
20   
21  #: Maximum number of contacts stored in a bucket; this should be an even number 
22  k = 8 
23   
24  #: Timeout for network operations (in seconds) 
25  rpcTimeout = 5 
26   
27  # Delay between iterations of iterative node lookups (for loose parallelism)  (in seconds) 
28  iterativeLookupDelay = rpcTimeout / 2 
29   
30  #: If a k-bucket has not been used for this amount of time, refresh it (in seconds) 
31  refreshTimeout = 3600 # 1 hour 
32  #: The interval at which nodes replicate (republish/refresh) data they are holding 
33  replicateInterval = refreshTimeout 
34  # The time it takes for data to expire in the network; the original publisher of the data 
35  # will also republish the data at this time if it is still valid 
36  dataExpireTimeout = 86400 # 24 hours 
37   
38  ######## IMPLEMENTATION-SPECIFIC CONSTANTS ########### 
39   
40  #: The interval in which the node should check its whether any buckets need refreshing, 
41  #: or whether any data needs to be republished (in seconds) 
42  checkRefreshInterval = refreshTimeout/5 
43   
44  #: Max size of a single UDP datagram, in bytes. If a message is larger than this, it will 
45  #: be spread accross several UDP packets. 
46  udpDatagramMaxSize = 8192 # 8 KB
